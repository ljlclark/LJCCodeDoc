public String GetIndentString()

Returns the current indent string.

Example

C#
// Defaults: IndentCharCount = 2, LineLimit = 80, WrapEnabled = false.
var hb = new HTMLBuilder();

hb.AddIndent(1);

// Example Method:
var result = hb.GetIndentString();
hb.AddText(result);

hb.AddText(":");
result = hb.ToString();

// result:
//   :
